        "#denoting rooms as 6 according to problem\n",
        "def set_initial_state(rooms =6):\n",
        "    \n",
        "    #Store the initial State\n",
        "    return np.random.randint(0, rooms)\n",
        "\n",
        "#initializing the action to be implement from the matrix\n",
        "def get_action(current_state, agent_mat):\n",
        "    \n",
        "    #Given a State, Choose from the Possiable\n",
        "    valid_actions = []\n",
        "    for action in enumerate(agent_mat[current_state]):\n",
        "        if action[1] !=-1:\n",
        "            valid_actions +=[action[0]]\n",
        "    return random.choice(valid_actions)\n",
        "\n",
        "def take_act(current_state, agent_mat,gamma, verbose = False):\n",
        "    \n",
        "    #Take a Single Action\n",
        "    action = get_action(current_state, agent_mat)\n",
        "    sa_reward= agent_mat[current_state,action] #Current State-action Reward\n",
        "    ns_reward= max(qmatrix[action,]) #Next State-action Reward\n",
        "    print(ns_reward)\n",
        "    q_cur_state = sa_reward + (gamma * ns_reward) #Current state will be updated with the actions done \n",
        "    qmatrix[current_state, action] = q_cur_state #  Mutates q Matrix\n",
        "    new_state = action\n",
        "    if verbose:\n",
        "        print(qmatrix)\n",
        "        print(f\"Old State: {current_state} | New State : {new_state}\\n\\n\")\n",
        "        if new_state == 5:   #When new State get to room 5 , Agent will stop the traversal \n",
        "            print(f\" Agent Has Reached their Goal !!! \")\n",
        "    return new_state"

        "def initial_episode(agent_mat,initial_state, gamma, verbose =False):\n",
        "    \n",
        "    # Runs 1 episode (or Simulation) until the agent reach it's goal-state\n",
        "    \n",
        "    current_state = initial_state\n",
        "    while True: # We won't use cureent state == 5 because possiable to initialize in 5\n",
        "        current_state = take_act(current_state, agent_mat, gamma , verbose)\n",
        "        if current_state == 5:\n",
        "            break\n",
        "                                    \n",
        "def train_agent(iteration , agent_mat , gamma , verbose: False): #Training the agent for q-learning\n",
        "        \n",
        "        #Runs a given number of episodes then normalize the matrix\n",
        "        print(\"Training in Progress>>>>\")\n",
        "        for episode in range(iteration):\n",
        "            initial_state = set_initial_state()\n",
        "            initial_episode(agent_mat, initial_state, gamma, verbose)\n",
        "        print(\"Training Complete! >>\")\n",
        "        \n",
        "        return qmatrix"
